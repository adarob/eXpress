00195 double BiasBoss::get_target_bias(std::vector<float>& start_bias, std::vector<float>& end_bias, const Target& targ) const
00196 {
00197     double tot_start = HUGE_VAL;
00198     double tot_end = HUGE_VAL;
00199     
00200     const Sequence& t_seq_fwd = targ.seq(0);
00201     const Sequence& t_seq_rev = targ.seq(1);
00202 
00203     for (size_t i = 0; i < targ.length(); ++i)
00204     {
00205         start_bias[i] = _5_seq_bias.get_weight(t_seq_fwd, i);
00206         end_bias[targ.length()-i-1] = _3_seq_bias.get_weight(t_seq_rev, i);
00207         tot_start = log_sum(tot_start, start_bias[i]);
00208         tot_end = log_sum(tot_start, end_bias[i]);
00209     }
00210     
00211     double avg_bias = (tot_start + tot_end) - (2*log((double)targ.length()));
00212     assert(!isnan(avg_bias));
00213     return avg_bias;
00214 }
